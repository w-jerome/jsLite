Plugin script

// plugins/light-opacity.js

(function () {
	$.fn.extend({

		/*
		* Opacity
		*/
		opacity: function ( params ){

			if ( typeof params != 'string' && typeof params != 'number' ) {
				return false;
			}
			for ( var i = 0; i < this.length; i++ ) {
				var element = this[i];
				element.style.opacity = params;
			}
		},
	});
})();
